import serial
import time

PORT = "/dev/ttyACM0"  # Windows: COM3, COM4, etc.
BAUD = 115200

def start_scan():
    print("[+] Scan started")

def stop_scan():
    print("[-] Scan stopped")

def snapshot():
    print("[*] Snapshot taken")

def main():
    ser = serial.Serial(PORT, BAUD, timeout=1)
    print("[*] Listening for Flipper commands...")

    while True:
        line = ser.readline().decode(errors="ignore").strip()
        if not line:
            continue

        print(f"[FLIPPER] {line}")

        if line == "SCAN_START":
            start_scan()
        elif line == "SCAN_STOP":
            stop_scan()
        elif line == "SNAPSHOT":
            snapshot()
        elif line == "EXIT":
            print("[!] Flipper requested exit")
            break

        time.sleep(0.05)

if __name__ == "__main__":
    main()
